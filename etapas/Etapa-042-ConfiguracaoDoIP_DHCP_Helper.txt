!O IP Helper (Ajuda de Endereço IP) são endereços IP configurados em uma Interface Roteada como uma Interface de VLAN
!ou uma Tnterface Ethernet (FasEthert, GigabitEthernet, etc) de roteadores que permite que esse dispositivo específico
!atue como um intermediário (middle man) que encaminha a solicitação DHCP do BOOTP (Broadcast) que recebe em uma interface
!para o Servidor DHCP especificado pelo endereço IP Helper via Unicast.

!O IP DHCP Client (Cliente do DHCP) é uma das formas de configuração do endereço IPv4 ou IPv6 em uma Interface de Rede, 
!esse recurso permite configuração uma interface dinâmicamente, obtendo o endereço IPv4 ou IPv6 dos servidores DHCP da Rede.
!Interface configuradas com o DHCP Client geralmente são utilizadas para se conectar com ISP (Internet Service Provider).

!TERCEIRA ETAPA: Configuração do Roteador 1941-3
!Acessando o modo EXEC Privilégiado
enable

	!Acessando o modo de Configuração Global de Comandos
	configure terminal

		!Configurando o hostname do Router 1941-3
		hostname dnsgoogle

		!Configurando a Interface de WAN do Router 1941-3
		interface gigabitEthernet 0/0
			description Interface de WAN do DNS do Google
			ip address 200.204.0.126 255.255.255.128

			!Configurando o Recurso de Ajuda de Endereço DHCP na Interface WAN
			!DICA: configurando o endereço IPv4 do Servidor de DHCP que possui o Escopo da Rede configurado
			!OBSERVAÇÃO: esse recurso funciona no Router ou Switch Layer, principalmente quando trabalhamos com VLAN
			ip helper-address 8.8.8.8
			no shutdown
			exit

		!Configurando a Interface de LAN do Router 1941-3
		interface gigabitEthernet 0/1
			description Interface de LAN do DNS do Google
			ip address 8.8.8.254 255.255.255.0
			no shutdown
			exit

		!Configurando a Rota Padrão de Saída do Router 1941-3
		!OBSERVAÇÃO: na vida real de Provedores de Serviços de Internet (ISP), não é configurado Rota Padrão de retorno
		!para o cliente, nesse cenário é utilizado NAT (Network Address Translation), que será visto futuramente nas 
		!configurações de acesso a internet
		ip route 0.0.0.0 0.0.0.0 GigabitEthernet0/0
		end
write

!QUARTA ETAPA: Configuração do Roteador 2911-1
enable
	configure terminal
		interface GigabitEthernet 0/1

			!Configurando a Interface do Router 2911-1 para obter endereço IPv4 via DHCP Client
			!DICA: interface configuradas para obter endereços IPv4 via DHCP geralmente são utilizadas para conexão com a internet
			!OBSERVAÇÃO: não é recomendado utilizar interfaces configuradas com DHCP Client na rede LAN, somente na WAN é em casos
			!especificos, quando se conectamos com o ISP (Internet Service Provider) ou Links Privados.
			ip address dhcp
			no shutdown
			exit

		!Configurando o Processo do Protocolo RIPv2
		router rip

			!Configurando a redistribuição da Rota Padrão para os outros roteadores do domínio do RIPv2
			!DICA: na vida real se torna complicado propagar e administrar rota estática padrão na rede, esse recurso parecido com
			!o comando: redistribute que facilita nossa vida informando o roteador na rede que tem a rota padrão para a Internet
			!OBSERVAÇÃO: cuidao com o comando default-information quando utilizado em roteadores ou switch layer 3 que tem roteadores
			!padrão para outra redes que não e da interface, utilizando o comando: ip route 0.0.0.0 0.0.0.0 gigabitEthernet 0/0
			!IMPORTANTE: a rota padrão desse roteador se aprendida e configurada dinâmicamente pelo DHCP Client.
			default-information originate
			end
write
