!Primeiro acesso no modo EXEC de Comandos >
switch>

!Acessando o modo EXEC Privilegiado #
enable

!Configuração de Data/Hora em inglês, você pode usar abreviado ou completo
!EXEMPLO: April ou Apr | December ou Dec
!EXEMPLO: Hora no formato: 00:00:00 - Data no formato: Dia Mês Ano
!DICA: Recomendado utilizar o Protocolo NTP (Network Time Protocol)
clock set 14:00:00 16 April 2020

	!Acessando o modo de configuração global de comandos
	configure terminal
  
	!Configuração do nome do switch
	!OBSERVAÇÃO: obrigatório para a configuração do SSH e demais serviços de redes
	!DICA: utilizar nomes curtos e objetivos, não usar caracteres especiais
	hostname sw-l2-2960-1
  
	!Habilitar o serviço de criptografia de senhas do Tipo-7 Password 
	!DICA: senhas do Tipo-7 por padrão não são criptografas
	!OBSERVAÇÃO: senhas do Tipo-7 são fáceis de serem quebradas
	service password-encryption
	
	!Habilitando o serviço de marcação de Data/Hora detalhado nos Logs
	!DICA: recurso utilizado em sistemas de monitoramento e desempenho de rede
	service timestamps log datetime msec
  
	!Desativar a resolução de nomes de domínio
	!DICA: desabilitar esse recurso resolve o problema de travamento de comandos
	!EXEMPLO: - para desbloquear você preciona: Ctrl+Shift+6
	!DICA: o comando: no, e usado para desabilitar ou remover configurações
	no ip domain-lookup
  
	!Configuração do banner da mensagem do dia
	!DICA: existe vários tipos de Banner no Cisco IOS, o MOTD e o mais usado
	!DICA: não utilizar acentuação e textos longos ou complexos
	!DESAFIO: Buscar na Internet imagens ASCII para o Banner MOTD
	!CUIDADO: utilizar o mesmo caracter de início e fim do Banner
	banner motd #AVISO: acesso autorizado somente a funcionarios#
  
	!Habilitando o uso senha do Tipo-5 Secret para acessar o modo EXEC Privilegiado
	!DICA: senhas do Tipo-5 por padrão são criptografas
	!DICA: por padrão e liberado sem segurança o acesso ao modo EXEC Privilegiado 
	enable secret vaamonde@pti
  
	!Criação dos usuários locais utilizando senhas do Tipo-5 e Tipo-7 e privilégios diferenciados
	!DICA: existe 16 níveis de privilégios no Cisco IOS (0 até 15)
	!DICA: não e recomendo criar usuários utilizando senhas do Tipo-7
	!DICA: existe a possibilidade de se autenticar no switch utilizando o RADIUS ou TACACS
	!Criação de usuários comuns para administrar o swtich, privilégio padrão 1
	username robson secret vaamonde@pti
	username vaamonde password vaamonde@pti
	!Criação de um super usuário, com todos os privilégios para administrar o switch
	!DICA: usuários como Privilégio 15 não precisa digitar o comando enable
	username admin privilege 15 secret vaamonde@pti
	
	!Acessando a linha console, porta padrão de acesso Out-of-Band (Fora da Banda)
	!DICA: conexão utilizando o Cabo Console RS232/DB9 ou USB e software de Console (PuTTY)
	!DICA: em switch temos apenas uma porta console RS232/DB9 ou USB
	line console 0
		
		!Forçando fazer login local utilizando usuário e senha locais do switch
		login local
		
		!Habilitando senha de acesso do Tipo-7 Password
		!DICA: em porta console não temos a opção de criar senhas do Tipo-5
		!OBSERVAÇÃO: porta console e considerada porta física não remota ou virtual
		!OBSERVAÇÃO: esse recurso só será utilizado, caso não exista usuários locais
		password vaamonde@pti
		
		!Sincronizando as mensagens de logs na tela
		!DICA: esse recurso facilita bastante, pois não atrapalha na digitação de comandos
		logging synchronous
		
		!Habilitando o tempo de inatividade do console
		!DICA: configuração de minutos e depois segundos
		!OBSERVAÇÃO: não e recomendo deixar o tempo muito curto e nem muito longo
		exec-timeout 5 30
		
		!Saindo de todos os níveis e voltando para o modo EXEC Privilégiado
		!DICA: você pode utilizar as teclas de atalho: Ctrl+C ou Ctrl+Z
		!DICA: o comando: exit sai nível por nível, o comando: end já sai de todos os níveis
		!OBSERVAÇÃO: os dois comandos são muito utilizado em scripts
		end

!Salvando as confgurações da memória RAM para a memória NVRAM
!DICA: você pode utilziar o comando: write, muito utilizado em scripts
copy running-config startup-config
	
!Visualizando as confgurações da memória RAM e NVRAM
show running-config
show startup-config
